GEDCOMNAME.SWIFT   
================================================================================================================
public typealias NameKey = String

public struct GedcomName: Comparable, CustomStringConvertible {
    var nameParts: [String]  // Name parts, before surname, surname, and after surname.
    var surnameIndex: Int?  // Index of surname if it exists.

    private init(nameParts: [String], surnameIndex: Int?) -- Private initializer.
    public init?(string: String) -- Public initializer from Gedcom-fromatted name string.
    public init?(from node: GedcomNode) -- Public initializer from either 0 INDI or 1 NAME node.
    public var description: String -- String description of this GedcomName.
    public var length: Int -- Return character lenth of name.
    public func displayName(upSurname: = false, surnameFirst:= false, limit:= 0)  -- Return formatted name string
    func format(surnameFirst: Bool) -> String -- [bad name] formats with surname first is flag true
    var firstInitial: Character? -- Return first initial of Gedcom name.
    var surname: String? -- Surname of this Gedcom name.
    var lowercaseGivens: [String] -- return array of givens in lower case.
    mutating func uppercaseSurname() -- Upper case the surname in situ.
    mutating func initialiseGiven(at i: Int) -- Convert a given to an initial in situ.
    mutating func removePart(at i: Int) -- Remove a name part (will not remove surname) in situ
    mutating func dropSuffix(range: Range<Int>) -- Remove suffixes.
    func compare(to other: GedcomName) -> ComparisonResult -- Compare two Gedcom names.
    public static func < (lhs: GedcomName, rhs: GedcomName) -> Bool -- Comparable protocol.
}

extension Person {
    public func displayName(upSurname: Bool, surnameFirst: Bool, limit: Int) -> String -- Return Person's display name.
}
